\section{First Aid and Medical}
\label{sec:medical}

\subsection{First Aid Kit}

\begin{tabular}{p{9cm} p{9cm}}
$\square$ Adhesive bandages (various sizes) & $\square$ Sterile gauze pads and rolls \\ 
$\square$ Medical tape & $\square$ Elastic bandage \\ 
$\square$ Antiseptic wipes & $\square$ Hydrogen peroxide or iodine \\ 
$\square$ Antibiotic ointment & $\square$ Burn cream \\ 
$\square$ Instant cold packs & $\square$ Tweezers \\ 
$\square$ Scissors & $\square$ Thermometer \\ 
$\square$ Disposable gloves & $\square$ CPR mask \\ 
\end{tabular}

\subsection{Medication and Health}

\begin{tabular}{p{9cm} p{9cm}}
$\square$ 7–30 day supply of prescriptions & $\square$ Copies of prescriptions \\ 
$\square$ Pain relievers (acetaminophen/ibuprofen) & $\square$ Antihistamines \\ 
$\square$ Anti-diarrheal medication & $\square$ Oral rehydration salts \\ 
$\square$ Inhalers (if applicable) & $\square$ EpiPen (if needed) \\ 
$\square$ Spare glasses/contact lenses & $\square$ Hearing aid batteries \\ 
\end{tabular}
